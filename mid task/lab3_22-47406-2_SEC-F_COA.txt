task:5)

.MODEL SMALL
.STACK 100H

.DATA
    prompt  DB 'Enter your password: $'
    success DB 0DH,0AH,'Password Matched$'
    fail    DB 0DH,0AH,'Password Not Matched$'

    password DB 'mypassword$'      
    input    DB 20 DUP('$')        

.CODE
MAIN PROC
    MOV AX, @DATA
    MOV DS, AX

start:
    
    MOV AH, 09H
    LEA DX, prompt
    INT 21H

    
    LEA DI, input     

read_input:
    MOV AH, 01H           
    INT 21H
    CMP AL, 0DH          
    JE compare_start
    STOSB                 
    JMP read_input

compare_start:
    MOV AL, '$'
    STOSB                

    
    LEA SI, password
    LEA DI, input

compare_loop:
    LODSB
    SCASB
    JNE not_matched     
    CMP AL, '$'
    JE matched           
    JMP compare_loop

matched:
    MOV AH, 09H
    LEA DX, success
    INT 21H
    JMP exit

not_matched:
    MOV AH, 09H
    LEA DX, fail
    INT 21H

exit:
    MOV AH, 4CH
    INT 21H

MAIN ENDP
END MAIN
